Concept

OpenGL does not manage windows or input. It is purely a rendering API. Without a window there is nowhere to draw, and without a context the GPU driver does not know which application's state to modify. GLFW solves the first problem: it provides a cross-platform API for creating OS windows and an attached OpenGL context. Once a window exists, calling glfwMakeContextCurrent binds that context to the current thread so every subsequent OpenGL call targets it.

OpenGL function addresses (pointers like glClear, glGenBuffers, etc.) are not statically linked. They are resolved at runtime from the GPU driver. GLEW queries the driver for every function the application might use and stores the addresses in global function pointers. Setting glewExperimental = GL_TRUE before calling glewInit forces GLEW to load all extensions even in a Core Profile context, which would otherwise cause some pointers to remain null.

The main loop runs once per frame. Each iteration clears the previous frame's pixels with glClear, renders the scene, then calls glfwSwapBuffers to flip the back buffer to the screen (double-buffering). Finally, glfwPollEvents processes any queued OS messages such as keyboard input or window resize. Calling glfwSwapInterval(1) before the loop enables V-Sync, which ties the swap to the display refresh rate and prevents screen tearing.

// file: src/main.cpp
#include <GL/glew.h>
#include <GLFW/glfw3.h>
#include <iostream>

int main() {
    // Initialise GLFW; abort immediately if the library fails to load
    if (!glfwInit()) {
        std::cerr << "Failed to initialize GLFW\n";
        return -1;
    }

    // Target OpenGL 4.3 Core Profile (required for compute shaders later)
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 4);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    // Create an 800x600 window; NULL = no fullscreen monitor, no shared context
    GLFWwindow* window = glfwCreateWindow(800, 600, "OpenGL Window", NULL, NULL);
    if (!window) {
        std::cerr << "Failed to create GLFW window\n";
        glfwTerminate();
        return -1;
    }

    // Bind the window's OpenGL context to the current thread
    glfwMakeContextCurrent(window);

    // Force GLEW to load all extension pointers even in Core Profile
    glewExperimental = GL_TRUE;
    if (glewInit() != GLEW_OK) {
        std::cerr << "Failed to initialize GLEW\n";
        glfwTerminate();
        return -1;
    }

    // Confirm the driver version the GPU is exposing
    std::cout << "OpenGL version: " << glGetString(GL_VERSION) << std::endl;

    // Cap frame rate to the monitor refresh rate (V-Sync)
    glfwSwapInterval(1);

    while (!glfwWindowShouldClose(window)) {
        // Set the colour the framebuffer is cleared to each frame
        glClearColor(0.1f, 0.1f, 0.15f, 1.0f);

        // Discard the previous frame's pixel data
        glClear(GL_COLOR_BUFFER_BIT);

        // Flip back buffer to screen
        glfwSwapBuffers(window);

        // Process keyboard, mouse, and window-close events
        glfwPollEvents();
    }

    // Release the window and all GLFW resources
    glfwDestroyWindow(window);
    glfwTerminate();
    return 0;
}
